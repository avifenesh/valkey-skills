## File 2: `01-cert-gen.yaml`

```yaml
apiVersion: v1
kind: ServiceAccount
metadata:
  name: valkey-cert-gen
  namespace: valkey
---
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: valkey-cert-gen
  namespace: valkey
rules:
  - apiGroups: [""]
    resources: ["secrets"]
    verbs: ["get", "create", "update", "patch"]
---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: valkey-cert-gen
  namespace: valkey
subjects:
  - kind: ServiceAccount
    name: valkey-cert-gen
    namespace: valkey
roleRef:
  kind: Role
  name: valkey-cert-gen
  apiGroup: rbac.authorization.k8s.io
---
apiVersion: batch/v1
kind: Job
metadata:
  name: valkey-cert-gen
  namespace: valkey
spec:
  ttlSecondsAfterFinished: 600
  backoffLimit: 2
  template:
    spec:
      serviceAccountName: valkey-cert-gen
      restartPolicy: OnFailure
      containers:
        - name: cert-gen
          image: alpine/k8s:1.29.2
          command:
            - sh
            - -c
            - |
              set -e

              # Skip if secret already exists
              if kubectl get secret valkey-tls -n valkey >/dev/null 2>&1; then
                echo "Secret valkey-tls already exists, skipping."
                exit 0
              fi

              apk add --no-cache openssl

              cd /tmp

              # CA
              openssl genrsa -out ca.key 4096
              openssl req -x509 -new -nodes -key ca.key -sha256 -days 3650 \
                -out ca.crt -subj "/CN=Valkey Cluster CA"

              # Server cert with SANs covering all pod DNS names
              cat > san.cnf <<EOF
              [req]
              req_extensions = v3_req
              distinguished_name = req_distinguished_name
              [req_distinguished_name]
              [v3_req]
              keyUsage = keyEncipherment, dataEncipherment
              extendedKeyUsage = serverAuth, clientAuth
              subjectAltName = @alt_names
              [alt_names]
              DNS.1 = *.valkey-headless.valkey.svc.cluster.local
              DNS.2 = valkey-headless.valkey.svc.cluster.local
              DNS.3 = valkey.valkey.svc.cluster.local
              DNS.4 = localhost
              IP.1  = 127.0.0.1
              EOF

              openssl genrsa -out server.key 2048
              openssl req -new -key server.key -out server.csr \
                -subj "/CN=valkey-cluster" \
                -config san.cnf
              openssl x509 -req -in server.csr \
                -CA ca.crt -CAkey ca.key -CAcreateserial \
                -out server.crt -days 365 -sha256 \
                -extensions v3_req -extfile san.cnf

              kubectl create secret generic valkey-tls \
                --from-file=ca.crt=ca.crt \
                --from-file=tls.crt=server.crt \
                --from-file=tls.key=server.key \
                -n valkey
              echo "TLS secret created."
```

## File 5: `04-cluster-init-job.yaml`

```yaml
# Idempotent: checks cluster state first, skips if already formed
apiVersion: batch/v1
kind: Job
metadata:
  name: valkey-cluster-init
  namespace: valkey
spec:
  ttlSecondsAfterFinished: 600
  backoffLimit: 5
  template:
    spec:
      restartPolicy: OnFailure
      containers:
        - name: cluster-init
          image: valkey/valkey-bundle:9
          env:
            - name: VALKEY_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: valkey-auth
                  key: password
          command:
            - sh
            - -c
            - |
              set -e

              NAMESPACE="valkey"
              HEADLESS="valkey-headless.${NAMESPACE}.svc.cluster.local"
              NODES=""
              for i in 0 1 2 3 4 5; do
                NODES="$NODES valkey-${i}.${HEADLESS}:6379"
              done

              CLI="valkey-cli --tls --cacert /etc/valkey/tls/ca.crt -a $VALKEY_PASSWORD"

              # Wait for all 6 pods to be reachable
              echo "Waiting for all nodes..."
              for i in 0 1 2 3 4 5; do
                HOST="valkey-${i}.${HEADLESS}"
                until $CLI -h "$HOST" ping | grep -q PONG; do
                  echo "  waiting for $HOST..."
                  sleep 3
                done
                echo "  $HOST: OK"
              done

              # Check if cluster already formed
              STATE=$($CLI -h valkey-0.${HEADLESS} cluster info | grep cluster_state | tr -d '\r')
              if echo "$STATE" | grep -q "cluster_state:ok"; then
                echo "Cluster already initialized and healthy. Exiting."
                exit 0
              fi

              echo "Forming cluster..."
              $CLI --cluster create $NODES \
                --cluster-replicas 1 \
                --cluster-yes

              echo "Verifying cluster..."
              sleep 3
              $CLI -h valkey-0.${HEADLESS} cluster info
              $CLI --cluster check valkey-0.${HEADLESS}:6379 \
                --tls --cacert /etc/valkey/tls/ca.crt -a "$VALKEY_PASSWORD"

              echo "Cluster initialization complete."
          volumeMounts:
            - name: valkey-tls
              mountPath: /etc/valkey/tls
              readOnly: true
      volumes:
        - name: valkey-tls
          secret:
            secretName: valkey-tls
            defaultMode: 0400
```

                echo "[ERROR] valkey-search module not loaded"

              echo "=== Step 2: Verify cluster state ==="

              echo ""
              echo "=== Step 3: Drop test index if it exists ==="
              $CLI FT.DROPINDEX test-products 2>/dev/null || true

              echo ""
              echo "=== Step 4: Create search index ==="
              $CLI FT.CREATE test-products \
                ON HASH PREFIX 1 product: \
                SCHEMA \
                  name TEXT \
                  category TAG \
                  price NUMERIC \
                  description TEXT
              echo "[OK] Index created"

              echo ""
              echo "=== Step 5: Insert test documents ==="
              $CLI HSET product:1 name "Valkey Database" category "software" price 0 \
                description "Fast in-memory data store with search support"
              $CLI HSET product:2 name "Redis Alternative" category "software" price 0 \
                description "Open source key-value database"
              $CLI HSET product:3 name "Search Module" category "software" price 0 \
                description "Vector and full-text search for Valkey"
              echo "[OK] 3 documents inserted"

              echo ""
              echo "=== Step 6: Full-text search ==="
              RESULT=$($CLI FT.SEARCH test-products "@name:Valkey")
              echo "$RESULT"
              if ! echo "$RESULT" | grep -q "product:1"; then
                echo "[ERROR] Full-text search did not return expected result"
                exit 1
              fi
              echo "[OK] Full-text search works"

              echo ""
              echo "=== Step 7: Tag + text combined query ==="
              RESULT=$($CLI FT.SEARCH test-products \
                "@category:{software} @description:search")
              echo "$RESULT"
              echo "[OK] Combined tag+text query works"

              echo ""
              echo "=== Step 8: Numeric range query ==="

## Deployment Guide

### Prerequisites

- Kubernetes 1.25+
- A `StorageClass` that supports `ReadWriteOnce` (update `storageClassName: standard` in file 4 to match your cluster)
- kubectl configured for your cluster

### 1. Set the password

```bash
echo -n 'your-secure-password' | base64
# paste output into 00-namespace-secrets.yaml -> data.password
```

### 2. Apply in order

```bash
kubectl apply -f 00-namespace-secrets.yaml
kubectl apply -f 01-cert-gen.yaml
# Wait for cert-gen job to complete before continuing
kubectl wait --for=condition=complete job/valkey-cert-gen -n valkey --timeout=120s
kubectl apply -f 02-configmap.yaml
kubectl apply -f 03-statefulset.yaml
# Wait for all 6 pods to be Running
kubectl rollout status statefulset/valkey -n valkey --timeout=300s
kubectl apply -f 04-cluster-init-job.yaml
kubectl wait --for=condition=complete job/valkey-cluster-init -n valkey --timeout=300s
```

### 3. Verify cluster

```bash
kubectl logs job/valkey-cluster-init -n valkey
```

Expected: `cluster_state:ok`, `cluster_slots_assigned:16384`, `cluster_size:3`

### 4. Run the search test

```bash
kubectl apply -f 05-test-job.yaml
kubectl wait --for=condition=complete job/valkey-search-test -n valkey --timeout=120s
kubectl logs job/valkey-search-test -n valkey
```

Expected: `All tests passed. valkey-search is operational.`

### 5. Connect from an application

```
valkeys://:your-password@valkey.valkey.svc.cluster.local:6379
```

Clients must handle `MOVED` and `ASK` redirects (cluster mode). Use a cluster-aware client: `iovalkey` (Node.js), `valkey-py` with cluster mode, `valkey-java` with `ValkeyClusterClient`, or any GLIDE client.

---

### Notes

**Module path**: The `valkey-bundle` image ships modules at `/opt/valkey/modules/valkeysearch.so`. If you use a custom image, confirm the path with `docker run --rm valkey/valkey-bundle:9 find / -name 'valkeysearch.so' 2>/dev/null`.

**StorageClass**: Replace `standard` with your provisioner (`gp3`, `premium-rwo`, `fast-ssd`, etc.). Set `allowVolumeExpansion: true` on the StorageClass to allow future PVC resizes.

**Certificate rotation**: The cert-gen Job creates certs valid for 365 days. Re-run the job (delete the old Secret first) annually, or replace the cert-gen Job with cert-manager `Certificate` resources for automatic rotation.

**Network Policy**: If your cluster enforces NetworkPolicy, allow TCP 6379 and 16379 between all `app: valkey` pods for the gossip bus.

**PVCs are not deleted** with the StatefulSet. Clean up manually with `kubectl delete pvc -l app=valkey -n valkey` only when decommissioning.
