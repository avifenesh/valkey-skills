## `examples/cache-aside-layer.ts`

```typescript
import {
    GlideClusterClient,
    TimeUnit,
    RequestError,
    ConnectionError,
    TimeoutError,
    ClosingError,
} from "@valkey/valkey-glide";

export interface CacheConfig {
    addresses: Array<{ host: string; port: number }>;
    requestTimeout?: number;
    defaultTtlSeconds?: number;
    entityTtls?: Record<string, number>;
    readFrom?: "primary" | "preferReplica";
    keyPrefix?: string;
}

export interface CacheResult<T> {
    data: T;
    source: "cache" | "database";
}

export class CacheAsideLayer {
    private client: GlideClusterClient | null = null;
    private connected = false;
    private readonly prefix: string;
    private readonly defaultTtl: number;
    private readonly entityTtls: Record<string, number>;
    private readonly config: CacheConfig;

    constructor(config: CacheConfig) {
        this.config = config;
        this.prefix = config.keyPrefix ?? "cache";
        this.defaultTtl = config.defaultTtlSeconds ?? 3600;
        this.entityTtls = config.entityTtls ?? {};
    }

    async connect(): Promise<void> {
        try {
            this.client = await GlideClusterClient.createClient({
                addresses: this.config.addresses,
                requestTimeout: this.config.requestTimeout ?? 3000,
                readFrom: this.config.readFrom ?? "preferReplica",
                connectionBackoff: {
                    numberOfRetries: 3,
                    factor: 500,
                    exponentBase: 2,
                    jitterPercent: 20,
                },
            });
            await this.client.ping();
            this.connected = true;
        } catch (err) {
            console.error("[WARN] Cache connection failed, running without cache:", err);
            this.client = null;
            this.connected = false;
        }
    }

    async disconnect(): Promise<void> {
        if (this.client) {
            this.client.close();
            this.client = null;
            this.connected = false;
        }
    }

    /**
     * Cache-aside read: try cache first, fall back to fetcher, populate cache on miss.
     * If fetcher returns null (entity not found), returns null without caching.
     */
    async getOrFetch<T>(
        entityType: string,
        id: string,
        fetcher: () => Promise<T | null>,
    ): Promise<CacheResult<T> | null> {
        const key = this.buildKey(entityType, id);

        if (this.connected && this.client) {
            try {
                const cached = await this.client.get(key);
                if (cached !== null) {
                    return { data: JSON.parse(cached as string), source: "cache" };
                }
            } catch (err) {
                this.handleError("get", err);
            }
        }

        const data = await fetcher();
        if (data === null) return null;

        // Populate cache - fire-and-forget, don't block the response
        this.setCache(key, entityType, data);

        return { data, source: "database" };
    }

    /** Populate cache after a create or update. */
    async populate<T>(entityType: string, id: string, data: T): Promise<void> {
        this.setCache(this.buildKey(entityType, id), entityType, data);
    }

    /** Invalidate a single entry. Call on update or delete. */
    async invalidate(entityType: string, id: string): Promise<void> {
        if (!this.connected || !this.client) return;
        try {
            await this.client.del([this.buildKey(entityType, id)]);
        } catch (err) {
            this.handleError("invalidate", err);
        }
    }

    async healthCheck(): Promise<boolean> {
        if (!this.connected || !this.client) return false;
        try {
            await this.client.ping();
            return true;
        } catch {
            this.connected = false;
            return false;
        }
    }

    isConnected(): boolean {
        return this.connected;
    }

    private buildKey(entityType: string, id: string): string {
        return `${this.prefix}:${entityType}:${id}`;
        // produces keys like: myapp:userProfile:abc123
    }

    private getTtl(entityType: string): number {
        return this.entityTtls[entityType] ?? this.defaultTtl;
    }

    private setCache<T>(key: string, entityType: string, data: T): void {
        if (!this.connected || !this.client) return;
        const ttl = this.getTtl(entityType);
        this.client
            .set(key, JSON.stringify(data), {
                expiry: { type: TimeUnit.Seconds, count: ttl },
            })
            .catch((err) => this.handleError("set", err));
    }

    private handleError(operation: string, err: unknown): void {
        if (err instanceof ClosingError || err instanceof ConnectionError) {
            this.connected = false;
            console.error(`[WARN] Cache ${operation}: connection lost -`, err);
        } else if (err instanceof TimeoutError) {
            console.error(`[WARN] Cache ${operation}: timeout`);
        } else if (err instanceof RequestError) {
            console.error(`[WARN] Cache ${operation}: ${err.message}`);
        } else {
            console.error(`[WARN] Cache ${operation}: unexpected error -`, err);
        }
    }
}

export async function createCacheLayer(config: CacheConfig): Promise<CacheAsideLayer> {
    const layer = new CacheAsideLayer(config);
    await layer.connect();
    return layer;
}
```

## `examples/cache-aside-usage.ts`

```typescript
import express, { Request, Response } from "express";
import { CacheAsideLayer, createCacheLayer } from "./cache-aside-layer";

interface UserProfile {
    id: string;
    email: string;
    displayName: string;
    bio?: string;
    updatedAt: number;
}

// --- Service layer wires cache + DB ---

class UserService {
    private static readonly ENTITY = "userProfile";

    constructor(
        private cache: CacheAsideLayer,
        private db: UserDatabase,
    ) {}

    async getProfile(id: string): Promise<UserProfile | null> {
        const result = await this.cache.getOrFetch(
            UserService.ENTITY,
            id,
            () => this.db.findById(id),
        );
        return result?.data ?? null;
    }

    async createProfile(profile: UserProfile): Promise<UserProfile> {
        const created = await this.db.insert(profile);
        await this.cache.populate(UserService.ENTITY, created.id, created);
        return created;
    }

    async updateProfile(id: string, fields: Partial<UserProfile>): Promise<UserProfile | null> {
        const updated = await this.db.update(id, fields);
        if (!updated) return null;
        // Invalidate then re-populate so the next read doesn't go to DB
        await this.cache.invalidate(UserService.ENTITY, id);
        await this.cache.populate(UserService.ENTITY, id, updated);
        return updated;
    }

    async deleteProfile(id: string): Promise<boolean> {
        const removed = await this.db.remove(id);
        if (removed) await this.cache.invalidate(UserService.ENTITY, id);
        return removed;
    }
}

// --- Express wiring ---

async function main() {
    const app = express();
    app.use(express.json());

    const cache = await createCacheLayer({
        addresses: [
            { host: "node1.example.com", port: 6379 },
            { host: "node2.example.com", port: 6380 },
            { host: "node3.example.com", port: 6381 },
        ],
        readFrom: "preferReplica",
        requestTimeout: 3000,
        defaultTtlSeconds: 3600,
        entityTtls: {
            userProfile: 1800,   // 30 min
            // sessionData: 900, // 15 min - add more entity types here
        },
        keyPrefix: "myapp",
    });

    const db = new UserDatabase();
    const users = new UserService(cache, db);

    app.get("/users/:id", async (req: Request, res: Response) => {
        const profile = await users.getProfile(req.params.id);
        if (!profile) { res.status(404).json({ error: "User not found" }); return; }
        res.json(profile);
    });

    app.post("/users", async (req: Request, res: Response) => {
        const { id, email, displayName, bio } = req.body;
        if (!id || !email || !displayName) {
            res.status(400).json({ error: "id, email, and displayName are required" }); return;
        }
        res.status(201).json(await users.createProfile({ id, email, displayName, bio, updatedAt: Date.now() }));
    });

    app.patch("/users/:id", async (req: Request, res: Response) => {
        const updated = await users.updateProfile(req.params.id, req.body);
        if (!updated) { res.status(404).json({ error: "User not found" }); return; }
        res.json(updated);
    });

    app.delete("/users/:id", async (req: Request, res: Response) => {
        const removed = await users.deleteProfile(req.params.id);
        if (!removed) { res.status(404).json({ error: "User not found" }); return; }
        res.status(204).end();
    });

    app.get("/health/cache", async (_req, res) => {
        const healthy = await cache.healthCheck();
        res.status(healthy ? 200 : 503).json({ cache: healthy ? "ok" : "unavailable" });
    });

    const server = app.listen(3000, () =>
        console.log(`[OK] :3000  cache=${cache.isConnected() ? "connected" : "unavailable"}`));

    const shutdown = async () => server.close(async () => { await cache.disconnect(); process.exit(0); });
    process.on("SIGTERM", shutdown);
    process.on("SIGINT", shutdown);
}

main().catch(console.error);
```

## Key design decisions

**Graceful degradation** - `connect()` catches all errors and sets `client = null`. Every read/write method guards `if (!this.connected || !this.client) return`. The app runs slower but doesn't crash.

**`handleError` marks the connection dead** - `ConnectionError` and `ClosingError` flip `this.connected = false` so subsequent commands skip straight to DB without attempting cache. Reconnection requires calling `connect()` again (e.g. from a health-check loop).

**Fire-and-forget cache writes** - `setCache` is private and never awaited. Cache population never adds latency to the response path; errors are logged only.

**Per-entity TTL** - `entityTtls` map takes precedence over `defaultTtlSeconds`. Add entries as needed (`sessionData: 900`, `productCatalog: 86400`, etc.).

**Key format** - `{keyPrefix}:{entityType}:{id}` (e.g. `myapp:userProfile:abc123`). The prefix isolates this app's namespace in a shared cluster.

**Update strategy** - On update, the service does `invalidate` then `populate`. This avoids a read-through on the next request while keeping the cache warm. There's a narrow race window between the two operations; for strict consistency requirements use a Lua script or transaction instead.

**`CacheResult<T>.source`** - The `"cache" | "database"` field lets you add observability (metrics, logging) without changing the service interface.
