## `generate-tls-secret.sh`

```bash
#!/usr/bin/env bash
# Run once before kubectl apply. Generates a wildcard TLS cert and stores it
# as Kubernetes Secrets in the valkey-cluster namespace.
set -euo pipefail

NAMESPACE=valkey-cluster
SERVICE=valkey-headless
CLUSTER_DOMAIN=cluster.local
DAYS=365
CERT_DIR=$(mktemp -d)
trap 'rm -rf "${CERT_DIR}"' EXIT

echo "[INFO] Creating namespace..."
kubectl create namespace "${NAMESPACE}" --dry-run=client -o yaml | kubectl apply -f -

echo "[INFO] Generating CA..."
openssl genrsa -out "${CERT_DIR}/ca.key" 4096
openssl req -x509 -new -nodes -key "${CERT_DIR}/ca.key" -sha256 -days "${DAYS}" \
  -out "${CERT_DIR}/ca.crt" \
  -subj "/CN=Valkey Cluster CA/O=valkey-cluster"

echo "[INFO] Generating server certificate..."
openssl genrsa -out "${CERT_DIR}/tls.key" 2048
openssl req -new -key "${CERT_DIR}/tls.key" -out "${CERT_DIR}/tls.csr" \
  -subj "/CN=valkey-cluster/O=valkey-cluster"

cat > "${CERT_DIR}/san.ext" <<EOF
[req]
req_extensions     = v3_req
distinguished_name = req_distinguished_name
[req_distinguished_name]
[v3_req]
subjectAltName = @alt_names
[alt_names]
DNS.1 = *.${SERVICE}.${NAMESPACE}.svc.${CLUSTER_DOMAIN}
DNS.2 = ${SERVICE}.${NAMESPACE}.svc.${CLUSTER_DOMAIN}
DNS.3 = valkey.${NAMESPACE}.svc.${CLUSTER_DOMAIN}
DNS.4 = localhost
IP.1  = 127.0.0.1
EOF

openssl x509 -req -in "${CERT_DIR}/tls.csr" \
  -CA "${CERT_DIR}/ca.crt" -CAkey "${CERT_DIR}/ca.key" -CAcreateserial \
  -out "${CERT_DIR}/tls.crt" -days "${DAYS}" -sha256 \
  -extensions v3_req -extfile "${CERT_DIR}/san.ext"

echo "[INFO] Storing TLS Secret..."
kubectl create secret generic valkey-tls \
  --namespace "${NAMESPACE}" \
  --from-file=ca.crt="${CERT_DIR}/ca.crt" \
  --from-file=tls.crt="${CERT_DIR}/tls.crt" \
  --from-file=tls.key="${CERT_DIR}/tls.key" \
  --dry-run=client -o yaml | kubectl apply -f -

echo "[INFO] Generating cluster password..."
VALKEY_PASSWORD=${VALKEY_PASSWORD:-$(openssl rand -base64 24 | tr -d '/+=')}
kubectl create secret generic valkey-auth \
  --namespace "${NAMESPACE}" \
  --from-literal=password="${VALKEY_PASSWORD}" \
  --dry-run=client -o yaml | kubectl apply -f -

echo "[OK] Done."
echo "[OK] To retrieve password: kubectl get secret valkey-auth -n ${NAMESPACE} -o jsonpath='{.data.password}' | base64 -d"
```

    tls-ciphers ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305

    loadmodule /usr/lib/valkey/modules/libvalkey_search.so

    dir /data
    appendonly yes
    appendfsync everysec
    auto-aof-rewrite-percentage 100
    auto-aof-rewrite-min-size 64mb

    save 300 100

    maxmemory-policy allkeys-lru

    lazyfree-lazy-eviction yes
    lazyfree-lazy-expire yes
    lazyfree-lazy-server-del yes
    lazyfree-lazy-user-del yes

    # --- General ---
    hz 15
    dynamic-hz yes
    loglevel notice
    logfile ""
    # Cluster bus port = client port + 10000 = 16379 (default, explicit here)
    cluster-port 16379

## `04-statefulset.yaml`

```yaml
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: valkey
  namespace: valkey-cluster
  labels:
    app: valkey
spec:
  serviceName: valkey-headless
  replicas: 6          # 3 primaries + 3 replicas; roles assigned by init job
  podManagementPolicy: Parallel
  updateStrategy:
    type: RollingUpdate
    rollingUpdate:
      partition: 0
  selector:
    matchLabels:
      app: valkey
  template:
    metadata:
      labels:
        app: valkey
    spec:
      serviceAccountName: valkey
      terminationGracePeriodSeconds: 60

      securityContext:
        runAsNonRoot: true
        runAsUser: 999
        fsGroup: 999

      affinity:
        podAntiAffinity:
          # Soft anti-affinity: prefer different nodes, but allow co-location
          # if the cluster doesn't have 6 separate nodes.
          preferredDuringSchedulingIgnoredDuringExecution:
            - weight: 100
              podAffinityTerm:
                labelSelector:
                  matchLabels:
                    app: valkey
                topologyKey: kubernetes.io/hostname

      initContainers:
        # Kernel tuning: THP and somaxconn.
        # Requires the kubelet to allow privileged init containers.
        # Remove if your cluster policy prohibits privileged pods, and apply
        # tuning via a DaemonSet at the node level instead.
        - name: disable-thp
          image: busybox:1.36
          command:
            - sh
            - -c
            - |
              echo never > /sys/kernel/mm/transparent_hugepage/enabled
              sysctl -w net.core.somaxconn=65535
              sysctl -w vm.overcommit_memory=1
          securityContext:
            privileged: true
          volumeMounts:
            - name: sys
              mountPath: /sys

      containers:
        - name: valkey
          image: valkey/valkey-bundle:9.0
          imagePullPolicy: IfNotPresent

          command: ["sh", "-c"]
          args:
            - |
              set -eu
              NAMESPACE=$(cat /var/run/secrets/kubernetes.io/serviceaccount/namespace)
              ANNOUNCE_HOST="${HOSTNAME}.valkey-headless.${NAMESPACE}.svc.cluster.local"
              exec valkey-server /etc/valkey/valkey.conf \
                --cluster-announce-hostname "${ANNOUNCE_HOST}" \
                --requirepass "${VALKEY_PASSWORD}" \
                --masterauth "${VALKEY_PASSWORD}"

          env:
            - name: VALKEY_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: valkey-auth
                  key: password

          ports:
            - name: client-tls
              containerPort: 6379
            - name: cluster-bus
              containerPort: 16379

          resources:
            requests:
              memory: 2Gi
              cpu: 500m
            limits:
              # 2x maxmemory (1500mb) to accommodate BGSAVE fork copy-on-write.
              # No CPU limit - CFS throttling causes latency spikes.
              memory: 3Gi

          # Startup probe: allow up to 5 minutes for AOF/RDB load on restart.
          startupProbe:
            exec:
              command:
                - sh
                - -c
                - |
                  valkey-cli --tls --no-auth-warning \
                    --cert /etc/valkey/tls/tls.crt \
                    --key  /etc/valkey/tls/tls.key \
                    --cacert /etc/valkey/tls/ca.crt \
                    -a "${VALKEY_PASSWORD}" ping
            initialDelaySeconds: 5
            periodSeconds: 10
            timeoutSeconds: 5
            failureThreshold: 30

          # Liveness probe: restart pod if completely unresponsive.
          livenessProbe:
            exec:
              command:
                - sh
                - -c
                - |
                  valkey-cli --tls --no-auth-warning \
                    --cert /etc/valkey/tls/tls.crt \
                    --key  /etc/valkey/tls/tls.key \
                    --cacert /etc/valkey/tls/ca.crt \
                    -a "${VALKEY_PASSWORD}" ping
            initialDelaySeconds: 30
            periodSeconds: 10
            timeoutSeconds: 5
            failureThreshold: 3

          # Readiness probe: gate traffic until the server is responding.
          # A simple PING is correct here - cluster_state:ok is not available
          # until after the init job runs, so checking it would block forever.
          readinessProbe:
            exec:
              command:
                - sh
                - -c
                - |
                  valkey-cli --tls --no-auth-warning \
                    --cert /etc/valkey/tls/tls.crt \
                    --key  /etc/valkey/tls/tls.key \
                    --cacert /etc/valkey/tls/ca.crt \
                    -a "${VALKEY_PASSWORD}" ping
            initialDelaySeconds: 10
            periodSeconds: 5
            timeoutSeconds: 3
            failureThreshold: 3

          volumeMounts:
            - name: valkey-data
              mountPath: /data
            - name: config
              mountPath: /etc/valkey
            - name: tls
              mountPath: /etc/valkey/tls
              readOnly: true

      volumes:
        - name: config
          configMap:
            name: valkey-config
        - name: tls
          secret:
            secretName: valkey-tls
        - name: sys
          hostPath:
            path: /sys

  volumeClaimTemplates:
    - metadata:
        name: valkey-data
      spec:
        accessModes: ["ReadWriteOnce"]
        # Replace with the StorageClass appropriate for your cluster:
        #   AWS EKS: gp3   GKE: premium-rwo   AKS: managed-premium
        storageClassName: fast-ssd
        resources:
          requests:
            # 4x maxmemory: RDB snapshot + rewrite temp file + AOF + headroom.
            storage: 10Gi
```

## `06-init-job.yaml`

```yaml
# Runs once after the StatefulSet pods are up. Idempotent: if the cluster is
# already initialized it exits cleanly without making any changes.
apiVersion: batch/v1
kind: Job
metadata:
  name: valkey-cluster-init
  namespace: valkey-cluster
  labels:
    app: valkey
    component: init
spec:
  backoffLimit: 6
  template:
    metadata:
      labels:
        app: valkey
        component: init
    spec:
      serviceAccountName: valkey
      restartPolicy: OnFailure

      containers:
        - name: cluster-init
          image: valkey/valkey-bundle:9.0
          imagePullPolicy: IfNotPresent
          command: ["sh", "-c"]
          args:
            - |
              set -euo pipefail
              NAMESPACE=$(cat /var/run/secrets/kubernetes.io/serviceaccount/namespace)
              SVC=valkey-headless

              TLS="--tls --no-auth-warning \
                --cert /etc/valkey/tls/tls.crt \
                --key  /etc/valkey/tls/tls.key \
                --cacert /etc/valkey/tls/ca.crt"
              AUTH="-a ${VALKEY_PASSWORD}"

              # Build node list: first 3 pods become primaries, last 3 replicas.
              NODE_LIST=""
              for i in 0 1 2 3 4 5; do
                HOST="valkey-${i}.${SVC}.${NAMESPACE}.svc.cluster.local"
                NODE_LIST="${NODE_LIST} ${HOST}:6379"
              done

              echo "[INFO] Waiting for all 6 nodes to respond..."
              for i in 0 1 2 3 4 5; do
                HOST="valkey-${i}.${SVC}.${NAMESPACE}.svc.cluster.local"
                ATTEMPTS=0
                until valkey-cli ${TLS} ${AUTH} -h "${HOST}" ping 2>/dev/null \
                        | grep -q PONG; do
                  ATTEMPTS=$((ATTEMPTS + 1))
                  if [ "${ATTEMPTS}" -gt 60 ]; then
                    echo "[ERROR] Timed out waiting for ${HOST}"
                    exit 1
                  fi
                  echo "[INFO] Waiting for ${HOST} (attempt ${ATTEMPTS})..."
                  sleep 5
                done
                echo "[OK] ${HOST} is responding"
              done

              FIRST="valkey-0.${SVC}.${NAMESPACE}.svc.cluster.local"

              echo "[INFO] Checking cluster state..."
              CLUSTER_INFO=$(valkey-cli ${TLS} ${AUTH} -h "${FIRST}" cluster info)
              CLUSTER_STATE=$(echo "${CLUSTER_INFO}" \
                | grep "^cluster_state:" | tr -d '[:space:]' | cut -d: -f2)

              if [ "${CLUSTER_STATE}" = "ok" ]; then
                KNOWN=$(echo "${CLUSTER_INFO}" \
                  | grep "^cluster_known_nodes:" | tr -d '[:space:]' | cut -d: -f2)
                echo "[OK] Cluster already initialized (known_nodes=${KNOWN}). Nothing to do."
                exit 0
              fi

              echo "[INFO] Creating cluster (3 primaries + 3 replicas)..."
              valkey-cli ${TLS} ${AUTH} \
                --cluster create ${NODE_LIST} \
                --cluster-replicas 1 \
                --cluster-yes

              echo "[INFO] Waiting for cluster to stabilize..."
              sleep 8

              echo "[INFO] Final cluster state:"
              valkey-cli ${TLS} ${AUTH} -h "${FIRST}" cluster info

              echo "[INFO] Node map:"
              valkey-cli ${TLS} ${AUTH} -h "${FIRST}" cluster nodes

              echo "[INFO] Full cluster check:"
              valkey-cli ${TLS} ${AUTH} --cluster check "${FIRST}:6379"

              echo "[OK] Cluster initialization complete"

          env:
            - name: VALKEY_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: valkey-auth
                  key: password

          volumeMounts:
            - name: tls
              mountPath: /etc/valkey/tls
              readOnly: true

      volumes:
        - name: tls
          secret:
            secretName: valkey-tls
```

## `07-test-job.yaml`

```yaml
# Smoke test: verifies search module is loaded, creates a full-text + tag +
# numeric index, seeds data, and runs 5 query types. Apply after init job
# completes and cluster_state is ok.
apiVersion: batch/v1
kind: Job
metadata:
  name: valkey-search-test
  namespace: valkey-cluster
  labels:
    app: valkey
    component: test
spec:
  backoffLimit: 3
  template:
    metadata:
      labels:
        app: valkey
        component: test
    spec:
      restartPolicy: OnFailure

      containers:
        - name: search-test
          image: valkey/valkey-bundle:9.0
          imagePullPolicy: IfNotPresent
          command: ["sh", "-c"]
          args:
            - |
              set -euo pipefail
              NAMESPACE=$(cat /var/run/secrets/kubernetes.io/serviceaccount/namespace)
              HOST="valkey-0.valkey-headless.${NAMESPACE}.svc.cluster.local"

              TLS="--tls --no-auth-warning \
                --cert /etc/valkey/tls/tls.crt \
                --key  /etc/valkey/tls/tls.key \
                --cacert /etc/valkey/tls/ca.crt"
              # -c enables cluster-mode redirect following (MOVED/ASK)
              CLI="valkey-cli -c ${TLS} -a ${VALKEY_PASSWORD} -h ${HOST}"

              # --- 1. Cluster health ---
              echo "[INFO] Checking cluster state..."
              STATE=$(${CLI} cluster info | grep "^cluster_state:" \
                | tr -d '[:space:]' | cut -d: -f2)
              if [ "${STATE}" != "ok" ]; then
                echo "[ERROR] cluster_state=${STATE}, expected ok"
                exit 1
              fi
              echo "[OK] cluster_state=ok"

              # --- 2. Module presence ---
              echo "[INFO] Checking valkey-search module..."
              if ! ${CLI} module list | grep -qi "search"; then
                echo "[ERROR] valkey-search not in module list"
                ${CLI} module list
                exit 1
              fi
              echo "[OK] valkey-search loaded"

              # --- 3. Seed data (all keys share hash tag {products} so they
              #         land on the same slot - required for PREFIX scan) ---
              echo "[INFO] Seeding test data..."
              ${CLI} DEL "doc:{products}:1" "doc:{products}:2" "doc:{products}:3" \
                > /dev/null 2>&1 || true

              ${CLI} HSET "doc:{products}:1" \
                title "Valkey In-Memory Database" \
                category "nosql" \
                price 0 \
                description "Open source in-memory data store"

              ${CLI} HSET "doc:{products}:2" \
                title "Valkey Search Module" \
                category "search" \
                price 0 \
                description "Full-text and vector search for Valkey"

              ${CLI} HSET "doc:{products}:3" \
                title "PostgreSQL Relational Database" \
                category "sql" \
                price 49 \
                description "Advanced open source relational database"
              echo "[OK] 3 documents seeded"

              # --- 4. Create index ---
              echo "[INFO] Dropping stale test index (if any)..."
              ${CLI} FT.DROPINDEX test-idx DD 2>/dev/null || true

              echo "[INFO] Creating index..."
              ${CLI} FT.CREATE test-idx \
                ON HASH PREFIX 1 "doc:{products}:" \
                SCHEMA \
                  title       TEXT \
                  description TEXT \
                  category    TAG \
                  price       NUMERIC

              echo "[INFO] Waiting for index backfill..."
              sleep 3

              echo "[INFO] FT.INFO:"
              ${CLI} FT.INFO test-idx

              # --- 5. Query tests ---
              PASS=0
              FAIL=0

              run_test() {
                local label="$1"; shift
                local result
                result=$("$@" 2>&1)
                if echo "${result}" | grep -qi "error\|ERR\|wrong"; then
                  echo "[FAIL] ${label}: ${result}"
                  FAIL=$((FAIL + 1))
                else
                  echo "[PASS] ${label}"
                  PASS=$((PASS + 1))
                fi
              }

              run_test "keyword search" \
                ${CLI} FT.SEARCH test-idx "@title:Valkey"

              run_test "phrase search" \
                ${CLI} FT.SEARCH test-idx "@title:\"In-Memory Database\""

              run_test "prefix search" \
                ${CLI} FT.SEARCH test-idx "@title:val*"

              run_test "tag filter" \
                ${CLI} FT.SEARCH test-idx "@category:{nosql}"

              run_test "numeric range - free items" \
                ${CLI} FT.SEARCH test-idx "@price:[0 0]"

              run_test "combined tag + numeric" \
                ${CLI} FT.SEARCH test-idx "@category:{sql} @price:[1 100]"

              run_test "FT._LIST" \
                ${CLI} FT._LIST

              echo ""
              echo "[INFO] Results: ${PASS} passed, ${FAIL} failed"
              if [ "${FAIL}" -gt 0 ]; then
                echo "[ERROR] ${FAIL} test(s) failed"
                exit 1
              fi
              echo "[OK] All search tests passed"

          env:
            - name: VALKEY_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: valkey-auth
                  key: password

          volumeMounts:
            - name: tls
              mountPath: /etc/valkey/tls
              readOnly: true

      volumes:
        - name: tls
          secret:
            secretName: valkey-tls
```

## Deployment Guide

### Prerequisites

- Kubernetes 1.25+
- `kubectl` configured for the target cluster
- A StorageClass that supports `ReadWriteOnce` (update `storageClassName` in the StatefulSet)
- Nodes that allow privileged init containers, or a DaemonSet applying the kernel tuning beforehand

### Step 1 - Generate TLS certs and secrets

```bash
chmod +x generate-tls-secret.sh
./generate-tls-secret.sh
```

This creates two Secrets in the `valkey-cluster` namespace: `valkey-tls` and `valkey-auth`. The password is printed to stdout - save it.

### Step 2 - Verify the module path

Before applying, confirm the path inside the image:

```bash
kubectl run --rm -it probe --image=valkey/valkey-bundle:9.0 \
  --restart=Never -- find /usr -name "libvalkey_search.so"
```

If the path differs from `/usr/lib/valkey/modules/libvalkey_search.so`, update the `loadmodule` line in `01-configmap.yaml`.

### Step 3 - Update StorageClass

Edit `04-statefulset.yaml` and replace `fast-ssd` with your cluster's SSD StorageClass:

| Platform | Class |
|----------|-------|
| AWS EKS | `gp3` |
| GKE | `premium-rwo` |
| AKS | `managed-premium` |
| On-prem (local) | `local-path` or your Ceph/Longhorn class |

### Step 4 - Apply manifests

```bash
kubectl apply -f 00-namespace.yaml
kubectl apply -f 01-configmap.yaml
kubectl apply -f 02-rbac.yaml
kubectl apply -f 03-services.yaml
kubectl apply -f 04-statefulset.yaml
kubectl apply -f 05-pdb.yaml
```

Wait for all 6 pods to reach Running state:

```bash
kubectl rollout status statefulset/valkey -n valkey-cluster
kubectl get pods -n valkey-cluster -w
```

### Step 5 - Initialize the cluster

```bash
kubectl apply -f 06-init-job.yaml
kubectl wait --for=condition=complete job/valkey-cluster-init \
  -n valkey-cluster --timeout=300s
kubectl logs job/valkey-cluster-init -n valkey-cluster
```

Expected tail of the logs:

```
[OK] cluster_state=ok
[OK] Cluster initialization complete
```

### Step 6 - Run the search smoke test

```bash
kubectl apply -f 07-test-job.yaml
kubectl wait --for=condition=complete job/valkey-search-test \
  -n valkey-cluster --timeout=120s
kubectl logs job/valkey-search-test -n valkey-cluster
```

Expected tail:

```
[OK] All search tests passed
```

### Ongoing operations

**Rolling update** - change the image tag in the StatefulSet and apply. The `RollingUpdate` strategy replaces one pod at a time. The PDB ensures at most one disruption at a time.

**Scale out** - increase `replicas` and re-run the init job (it is idempotent and will skip if already initialized); then run `valkey-cli --cluster rebalance` to assign slots to new primaries.

**Certificate rotation** - update the `valkey-tls` Secret with new cert material. `tls-auto-reload-interval 3600` causes each node to pick up the new cert within an hour without restart. To force immediate reload: `CONFIG SET tls-auto-reload-interval 1` on each node, wait, then set it back.

**PVC cleanup** - PVCs are not deleted with the StatefulSet. When decommissioning: `kubectl delete pvc -l app=valkey -n valkey-cluster`.

---

### Key design notes

| Decision | Rationale |
|----------|-----------|
| `cluster-preferred-endpoint-type hostname` | Nodes gossip their DNS hostname instead of pod IP, so TLS cert validation succeeds against the wildcard SAN |
| `tls-auth-clients optional` | Application clients don't need client certs; inter-node mTLS uses the shared wildcard cert |
| `publishNotReadyAddresses: true` on headless Service | DNS records register immediately so pods can reach each other before readiness passes |
| No CPU limit | Kubernetes CFS CPU throttling causes latency spikes; set requests only |
| Memory limit = 2x maxmemory | BGSAVE fork uses copy-on-write and can temporarily double resident memory |
| `podManagementPolicy: Parallel` | All 6 pods start concurrently; init job handles sequencing |
